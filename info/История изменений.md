**Правка 1**
```python
def goto(self, x, y, z, yaw=0, wait=3):
    self.drone.go_to_local_point(...)
    time.sleep(wait)
```
В методе goto все сейчас работает по принципу отправил команду → подождал N секунд → отправил следующую.
Чтобы понять почему ломаются маршруты можно посмотреть по шагам, что происходит(для square):

Шаг 1: отправил (s, s), ждёт 3 сек

Шаг 2: дрон ещё летит к (s, s), код уже отправляет (-s, s)

Решение: встроенный метод point_reached()

```python
def goto(self, x, y, z, yaw=0, timeout=15):
    self._require_connection()

    self.drone.go_to_local_point(x=x, y=y, z=z, yaw_angle=yaw)

    start = time.time()
    while time.time() - start < timeout:
        try:
            if self.drone.point_reached():
                return True
        except Exception:
            pass
        time.sleep(0.1)

    print(f"[{self.drone_id}] Point timeout: {x,y,z}")
    return False
```

```python
def make_log(self):
    t = time.time() - (self.experiment_start_time or time.time())

    telemetry = self.get_telemetry()

    pos = telemetry["position"]
    vel = telemetry["velocity"]
    att = telemetry["attitude"]

    accel = telemetry["accel"]
    gyro = telemetry["gyro"]
    mag = telemetry["mag"]
    rpm = telemetry["rpm"]
    battery = telemetry["battery"]

    log = {
        "timestamp": time.time(),
        "t": t,

        "drone_id": self.drone_id,
        "experiment_id": self.experiment_id,
        "pattern": self.pattern,

        # position
        "x": pos["x"],
        "y": pos["y"],
        "z": pos["z"],

        # velocity
        "vx": vel["vx"],
        "vy": vel["vy"],
        "vz": vel["vz"],

        # attitude
        "roll": att["roll"],
        "pitch": att["pitch"],
        "yaw": att["yaw"],

        # imu
        "ax": accel[0] if accel else None,
        "ay": accel[1] if accel else None,
        "az": accel[2] if accel else None,

        "gx": gyro[0] if gyro else None,
        "gy": gyro[1] if gyro else None,
        "gz": gyro[2] if gyro else None,

        "mx": mag[0] if mag else None,
        "my": mag[1] if mag else None,
        "mz": mag[2] if mag else None,

        # motors
        "rpm1": rpm[0] if rpm else None,
        "rpm2": rpm[1] if rpm else None,
        "rpm3": rpm[2] if rpm else None,
        "rpm4": rpm[3] if rpm else None,

        # battery
        "battery": battery[0] if battery else None,
    }

    self.logger.write(log)
    return log
```
